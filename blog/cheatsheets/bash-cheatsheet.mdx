## Execute a command inside subfolders

```bash
function recursive_for_loop { 
    ls -1| until ! read f; do
        if [ -d $f  -a ! -h $f ];
        then
            cd -- "$f";
            recursive_for_loop
            echo "### Folder $f"; 
            git status; # replace here your command

            # use recursion to navigate the entire tree; git status;
            cd ..;
        fi;
    done;
};
recursive_for_loop
```
